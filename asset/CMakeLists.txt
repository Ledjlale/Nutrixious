
file(GLOB files LIST_DIRECTORIES false  icons/*)
foreach(f ${files})
	get_filename_component(filename  ${f} NAME)
	list(APPEND _APP_RC_FILES asset/icons/${filename})
endforeach()
set(_APP_RC_FILES ${_APP_RC_FILES} PARENT_SCOPE)
file(COPY android DESTINATION ${CMAKE_BINARY_DIR}/asset)
if(CMAKE_BUILD_TYPE STREQUAL "Debug")
	set(ANDROID_DEBUGGABLE "android:debuggable=\"true\"")
else()
	set(ANDROID_DEBUGGABLE)
endif()
configure_file(AndroidManifest.xml.in ${CMAKE_BINARY_DIR}/asset/android/AndroidManifest.xml @ONLY)
